ALTER PROCEDURE dbo.AddAuthor 
@id varchar(11), 
@fname varchar(20), 
@lname varchar(40), 
@phone char(12), 
@contract bit,
@returnID varchar(11) OUTPUT

AS

BEGIN
INSERT INTO authors
                         (au_id, au_lname, au_fname, phone, contract)
VALUES        (@id, @fname, @lname, @phone, @contract)
END


BEGIN
SELECT @returnID = authors.au_id from authors where @id = authors.au_id
END
